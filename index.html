comandos de git

git status // para ver el estado de nuestros canbios

git add . // para añadir los cambios

git commit -m "aquí la descripcion de los cambios"

git log // para ver el historial de cambios y commits hechos

git log --stat // para ver el historial y una estadistica de los cambios introducidos

git log --oneline // para ver todos los commits en una sola linea

git log --graph // para ver los cambios en una grafica

git log --oneline --graph --all // ara poner en una linea una grafica de todo

git diff // para ver diferencias entre los commits y los que ya hay grabados

git diff HEAD // para ver cambios introducidos en el Head de la pagina

git checkuot + el numero de la version a la que quieres regresar /// ojo tener cuidado con este comando ya que una vez ejecutado las versiones borradas no se pueden recuperar

git branch // para ver versiones del documento

git checkuot + el nombre de la version para cambiar entre versiones

git merge // para unir versiones del documento y hacer uno solo

git checkuot -b + nombre /// para generar una copia del documento

git branch -d + el nombre de la rama // para borrar la copia del documento si asi lo quieres
